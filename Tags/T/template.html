<template> 
<!DOCTYPE html>
<html>
    <head></head>
    <body>
        <button onclick="t()">Click</button>
        <template>
            Hidden content in template
        </template>
        <script>
            function t(){
                var temp = document.getElementsByTagName("template")[0];
                var clon = temp.content.cloneNode(true);
                let x = document.querySelector("template");
                x.appendChild(clon);
                x.style.display="block";
            }
        </script>
    </body>
</html>
